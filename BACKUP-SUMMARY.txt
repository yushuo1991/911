从本地恢复:

  cd ../restore
  npm install
  npm run dev

从GitHub恢复:
  git clone --branch v4.14-stable-20251002 https://github.com/yushuo1991/911.git
  cd 911
  npm install
  npm run dev

服务器部署:
  cd /www/wwwroot/stock-tracker
  git pull origin main
  docker compose down
  docker compose build --no-cache
  docker compose up -d
